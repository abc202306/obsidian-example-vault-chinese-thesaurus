```dataviewjs
/** @function */
const getGroupKey = await new Promise(resolve => dv.view("dvmodule/get-group-key", resolve))


const headers = [
	"File",
	"03 英文",
	"05 同义词",
	"06 上位词",
	"07 下位词",
	"08 相关词",
	"31 分类",
	"来源",
	"file.cday"
];

class ArrayUtil {
	static maxLenLinkArr = 3;
	static clipArr(arr){
		if(arr.length<=this.maxLenLinkArr){
			return arr;
		}
		return [
			...arr.slice(0, this.maxLenLinkArr), 
			"..."
		];
	}
	static getDescriptorItemLinkArr(linkArr){
		return linkArr
			.filter(l=>dv.func.contains(l.path,"/"))
			.map(l=>dv.func.link(
				l.path,
				"🟢"+l.path.replace(
					/(^.*\/)|(\.md$)/g,
					""
				)
			)).concat(linkArr
				.filter(l=>!l.path.includes("/"))
			);
	}
}

class PageProxy {
	#p;
	
	constructor(p){this.#p=p;}
	static get(p){return new this(p)}
	
	getFileLink(){
		return dv.func.link(
			this.#p.file.path, 
			"🟢"+this.#p.file.name
		);
	}
	getEnglish(){
		return this.#p["主题词-英文"];
	}
	getSynonyms(){
		return ArrayUtil.clipArr(
			this.#p["主题词-同义词"] || []
		)
	}
	getBroadTerms(){
		return (this.#p["主题词-上位词"] || []).map(l=>
			dv.func.link(
				l.path,
				"🟢"+l.path.replace(
					/(^.*\/)|(\.md$)/g,
					""
				)
			)
		)
	}
	getNarrowerTerms(){
		return ArrayUtil.clipArr(
			ArrayUtil.getDescriptorItemLinkArr(
				this.#p["主题词-下位词"] || []
			)
		)
	}
	getRelatedTerms(){
		return ArrayUtil.clipArr(
			ArrayUtil.getDescriptorItemLinkArr(
				this.#p["主题词-相关词"] || []
			)
		)
	}
	getClassifications(){
		return (this.#p["主题词-分类"] || []).map(l=>
			dv.func.link(
				l.path,
				"📗"+l.path.replace(
					/(^.*\/)|(\s主题词表\.md$)/g,
					""
				)
			)
		)
	}
	getSources(){
		return this.#p["主题词-来源"] || []
	}
	getCDay(){
		return this.#p.file.cday;
	}

	getElem(){
		return [
			this.getFileLink(),
			this.getEnglish(),
			this.getSynonyms(),
			this.getBroadTerms(),
			this.getNarrowerTerms(),
			this.getRelatedTerms(),
			this.getClassifications(),
			this.getSources(),
			this.getCDay()
		]
	}
}

function show(){
	dv.container.style.overflowX = "visible";

	const groups = dv
		.pages(`"${dv.current()["主题词 - 文件夹路径"]}"`)
		.sort(p=>p.file.ctime, "desc")
		.groupBy(p=>getGroupKey(p));
	
	groups.forEach(g=>{
		dv.header(4, g.key)
		dv.api.table(
			headers, 
			g.rows.map(p=>PageProxy.get(p).getElem()),
			dv.container.createDiv({attr:{style:"overflow:scroll;"}}),
			dv.component,
			dv.currentFilePath
		)
	});
}

show()

```
